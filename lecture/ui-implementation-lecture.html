Overview

全体方針

全てのUI環境に共通するのは「GAS WebAppのURLにHTTP POSTを投げて回答を受け取る」という構造。UIの見た目と実装言語は環境ごとに異なるが、コアロジックは同一。

UI環境参照RAG実装技術難易度
Unity エディタ CloudRAG C# EditorWindow ★★☆
Houdini パネル CloudRAG Python + PySide2 ★★☆
Webアプリ CloudRAG Next.js → Vercel / GitHub Pages ★☆☆
Desktop（Tauri） LocalRAG Rust + Webview ★★★
Claude Desktop LocalRAG MCP設定のみ（実装不要） ★☆☆

なぜGemini UIから直接呼べないのか

gemini.google.com は外部エンドポイントをツールとして登録する機能を持たない。GAS WebAppを中間層に置くことで、どのUI環境からも同一の呼び出しインターフェースが使える。

共通フロー

各UI環境
  │
  │  POST { query, dbKey }
  ▼
GAS WebApp  ──→  Notion API（指定DB検索）
  │                    │ 上位5件のsummary
  │◀───────────────────┘
  │
  │  context + query
  ▼
Gemini API（gemini-2.0-flash）
  │
  │  { answer: "..." }
  ▼
各UI環境に返す

RAGChatWindow.cs
using UnityEditor;
using System.Collections.Generic;

public class RAGChatWindow : EditorWindow
{
    private const string GAS_ENDPOINT =
        "https://script.google.com/macros/s/YOUR_ID/exec";

    private static readonly string[] DB_LABELS = {
        "Tool Docs（Unity / Houdini / DX12）",
        "Game Info（ゲーム情報）",
        "Research（論文・技術記事）",
        "Team Notes（ゼミ・議事録）"
    };
    private static readonly string[] DB_KEYS = {
        "tool_docs", "game_info", "research", "team_notes"
    };

    private int _selectedDb = 0;
    private string _inputText = "";
    private Vector2 _scrollPos;
    private readonly List<(string role, string text)> _messages = new();
    private bool _isLoading = false;
    private RAGClient _client;

    [MenuItem("Tools/RAG Chat")]
    public static void Open() => GetWindow<RAGChatWindow>("RAG Chat");

    private void OnEnable()
        => _client = new RAGClient(GAS_ENDPOINT);

    private void OnGUI()
    {
        // DB選択ドロップダウン
        _selectedDb = EditorGUILayout.Popup(_selectedDb, DB_LABELS);

        // チャット履歴スクロール表示
        _scrollPos = EditorGUILayout.BeginScrollView(_scrollPos);
        foreach (var (role, text) in _messages)
            EditorGUILayout.LabelField($"[{role}] {text}");
        EditorGUILayout.EndScrollView();

        // 入力欄と送信ボタン
        EditorGUILayout.BeginHorizontal();
        GUI.enabled = !_isLoading;
        _inputText = EditorGUILayout.TextField(_inputText);
        if (GUILayout.Button(_isLoading ? "..." : "送信", GUILayout.Width(60)))
            SendQuery();
        GUI.enabled = true;
        EditorGUILayout.EndHorizontal();
    }

    private async void SendQuery()
    {
        var query = _inputText.Trim();
        _inputText = "";
        _messages.Add(("user", query));
        _isLoading = true;
        Repaint();

        var answer = await _client.QueryAsync(query, DB_KEYS[_selectedDb]);
        _messages.Add(("assistant", answer));
        _isLoading = false;
        Repaint();
    }
}

RAGClient.cs

RAGClient.cs
using System.Net.Http;
using System.Text;
using System.Threading.Tasks;
using Newtonsoft.Json;

public class RAGClient
{
    private readonly string _endpoint;
    private static readonly HttpClient Http = new();

    public RAGClient(string endpoint) => _endpoint = endpoint;

    public async Task<string> QueryAsync(string query, string dbKey)
    {
        try
        {
            var body = JsonConvert.SerializeObject(new { query, dbKey });
            var content = new StringContent(body, Encoding.UTF8, "application/json");
            var res = await Http.PostAsync(_endpoint, content);
            var json = await res.Content.ReadAsStringAsync();
            dynamic data = JsonConvert.DeserializeObject(json);
            return (string)data?.answer ?? "回答を取得できませんでした";
        }
        catch (System.Exception e)
        {
            return $"エラー: {e.Message}";
        }
    }
}

panel.py
import hou
import urllib.request, json, threading
from PySide2 import QtWidgets, QtCore

GAS_ENDPOINT = "https://script.google.com/macros/s/YOUR_ID/exec"
DB_OPTIONS = {
    "Tool Docs（Houdini / Unity / DX12）": "tool_docs",
    "Game Info":                          "game_info",
    "Research":                           "research",
    "Team Notes":                         "team_notes",
}

class RAGChatPanel(QtWidgets.QDialog):
    answer_received = QtCore.Signal(str)

    def __init__(self):
        super().__init__(hou.qt.mainWindow())
        self.setWindowTitle("RAG Chat — Houdini")
        self.setMinimumSize(480, 560)
        self.setWindowFlags(
            QtCore.Qt.Tool | QtCore.Qt.WindowStaysOnTopHint
        )
        self._build_ui()
        self.answer_received.connect(self._on_answer)

    def _build_ui(self):
        layout = QtWidgets.QVBoxLayout(self)

        # DB選択
        self.db_combo = QtWidgets.QComboBox()
        for label in DB_OPTIONS:
            self.db_combo.addItem(label)
        layout.addWidget(self.db_combo)

        # チャット表示
        self.chat_view = QtWidgets.QTextEdit()
        self.chat_view.setReadOnly(True)
        self.chat_view.setStyleSheet(
            "background:#1a1a2e;color:#e0e0e0;font-size:13px;"
        )
        layout.addWidget(self.chat_view)

        # 入力欄 + 送信ボタン
        row = QtWidgets.QHBoxLayout()
        self.input_field = QtWidgets.QLineEdit()
        self.input_field.returnPressed.connect(self._send)
        row.addWidget(self.input_field)
        btn = QtWidgets.QPushButton("送信")
        btn.clicked.connect(self._send)
        row.addWidget(btn)
        layout.addLayout(row)

    def _send(self):
        query = self.input_field.text().strip()
        if not query: return
        db_key = DB_OPTIONS[self.db_combo.currentText()]
        self.input_field.clear()
        self._append("あなた", query, "#4ea8de")

        # 別スレッドでHTTPリクエスト（UIブロックしない）
        threading.Thread(
            target=self._fetch, args=(query, db_key), daemon=True
        ).start()

    def _fetch(self, query, db_key):
        try:
            payload = json.dumps({"query": query, "dbKey": db_key}).encode()
            req = urllib.request.Request(
                GAS_ENDPOINT, data=payload,
                headers={"Content-Type": "application/json"}
            )
            with urllib.request.urlopen(req, timeout=30) as r:
                data = json.loads(r.read().decode())
                self.answer_received.emit(data["answer"])
        except Exception as e:
            self.answer_received.emit(f"エラー: {e}")

    def _on_answer(self, text):
        self._append("RAG", text, "#3fb950")

    def _append(self, role, text, color):
        self.chat_view.append(
            f'<span style="color:{color};font-weight:bold;">[{role}]</span> {text}<br>'
        )

# シングルトン管理
_panel = None
def show():
    global _panel
    if _panel is None or not _panel.isVisible():
        _panel = RAGChatPanel()
    _panel.show()
    _panel.raise_()

app/api/chat/route.ts — GASプロキシ

なぜプロキシが必要か

GAS WebAppはCORSヘッダーの設定が限定的で、ブラウザから直接POSTするとエラーになるケースがある。Next.jsのAPI Routeをプロキシにすることで回避できる。

app/api/chat/route.ts
import { NextRequest, NextResponse } from 'next/server'

const GAS_ENDPOINT = process.env.GAS_ENDPOINT!

export async function POST(req: NextRequest) {
  const body = await req.json()
  try {
    const res = await fetch(GAS_ENDPOINT, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
    })
    const data = await res.json()
    return NextResponse.json(data)
  } catch (e) {
    return NextResponse.json(
      { answer: `エラー: ${e}`, status: 'error' },
      { status: 500 }
    )
  }
}

components/ChatWindow.tsx

components/ChatWindow.tsx
'use client'
import { useState, useRef, useEffect } from 'react'

const DB_OPTIONS = [
  { label: 'Tool Docs（Unity / Houdini / DX12）', value: 'tool_docs' },
  { label: 'Game Info（ゲーム情報）',              value: 'game_info' },
  { label: 'Research（論文・技術記事）',           value: 'research'  },
  { label: 'Team Notes（ゼミ・議事録）',           value: 'team_notes'},
]

type Msg = { role: 'user' | 'assistant'; text: string }

export default function ChatWindow() {
  const [messages, setMessages] = useState<Msg[]>([])
  const [input, setInput]       = useState('')
  const [dbKey, setDbKey]       = useState('tool_docs')
  const [loading, setLoading]   = useState(false)
  const bottomRef               = useRef<HTMLDivElement>(null)

  useEffect(() => {
    bottomRef.current?.scrollIntoView({ behavior: 'smooth' })
  }, [messages])

  const send = async () => {
    if (!input.trim() || loading) return
    const query = input.trim()
    setInput('')
    setMessages(prev => [...prev, { role: 'user', text: query }])
    setLoading(true)

    const res  = await fetch('/api/chat', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ query, dbKey }),
    })
    const data = await res.json()
    setMessages(prev => [...prev, { role: 'assistant', text: data.answer }])
    setLoading(false)
  }

  return (
    <div className="flex flex-col h-screen max-w-2xl mx-auto p-4">
      <select value={dbKey} onChange={e => setDbKey(e.target.value)}
        className="mb-3 p-2 rounded border text-sm">
        {DB_OPTIONS.map(o => <option key={o.value} value={o.value}>{o.label}</option>)}
      </select>
      <div className="flex-1 overflow-y-auto space-y-3 mb-3">
        {messages.map((m, i) => (
          <div key={i} className={`flex ${m.role==='user'?'justify-end':'justify-start'}`}>
            <div className={`max-w-[80%] px-3 py-2 rounded-xl text-sm whitespace-pre-wrap
              ${m.role==='user'?'bg-blue-600 text-white':'bg-gray-100 text-gray-800'}`}>
              {m.text}
            </div>
          </div>
        ))}
        {loading && <div className="flex justify-start">
          <div className="bg-gray-100 px-3 py-2 rounded-xl text-sm text-gray-400">考え中...</div>
        </div>}
        <div ref={bottomRef} />
      </div>
      <div className="flex gap-2">
        <input value={input} onChange={e => setInput(e.target.value)}
          onKeyDown={e => e.key==='Enter' && !e.shiftKey && send()}
          placeholder="質問を入力..." className="flex-1 px-3 py-2 rounded border text-sm"/>
        <button onClick={send} disabled={loading}
          className="px-4 py-2 bg-blue-600 text-white rounded text-sm disabled:opacity-50">
          送信
        </button>
      </div>
    </div>
  )
}

use serde::{Deserialize, Serialize};
use tauri::command;

#[derive(Serialize)]
struct RagRequest { query: String, db_key: String }

#[derive(Deserialize)]
struct RagResponse { answer: String }

// CloudRAG 呼び出し（GAS WebApp）
#[command]
pub async fn query_cloud_rag(query: String, db_key: String)
    -> Result<String, String>
{
    let client = reqwest::Client::new();
    let endpoint = "https://script.google.com/macros/s/YOUR_ID/exec";

    let res = client
        .post(endpoint)
        .json(&RagRequest { query, db_key })
        .send().await
        .map_err(|e| e.to_string())?;

    let data: RagResponse = res.json().await
        .map_err(|e| e.to_string())?;
    Ok(data.answer)
}

// LocalRAG 呼び出し（実際は rag_local_bridge.py の HTTP /query エンドポイントを叩くのが現行の標準経路。
// ここでは CLI 越しに直接呼ぶ代替パターンの例として、rag_service.py をワンライナーで叩く形を示す）
#[command]
pub async fn query_local_rag(query: String, namespace: String)
    -> Result<String, String>
{
    let cmd = format!(
        "uv run --directory C:/Users/YOUR_USERNAME/DevelopmentRAGEnvironment \
         python -c \"import sys; sys.path.insert(0,'scripts'); from rag_service import create_rag_service_from_env; \
         print(create_rag_service_from_env().search('{}', limit=5))\"",
        query.replace("'", "\\'")
    );
    let out = tokio::process::Command::new("wsl")
        .args(["bash", "-c", &cmd])
        .output().await
        .map_err(|e| e.to_string())?;

    Ok(String::from_utf8_lossy(&out.stdout).to_string())
}

JavaScript — レスポンスからバッジを描画

GAS WebApp クライアント JS
function renderSources(sources, extractionRate, extractionDetail) {
  const pct = Math.round((extractionRate || 0) * 100);
  const cited = (extractionDetail || []).filter(Boolean).length;

  // ヘッダー（抽出度サマリー）
  headerEl.innerHTML =
    `📎 参考情報 ${sources.length}件   ` +
    `<span class="extract-summary">💡 抽出度: ${cited}/${sources.length} (${pct}%) ▾</span>`;

  // プログレスバー
  barEl.innerHTML =
    `<div class="extract-fill" style="width:${pct}%"></div>`;

  // 各ソースにバッジを付与
  sources.forEach((src, i) => {
    const isCited = extractionDetail?.[i] ?? false;
    const badgeClass = isCited ? 'src-cited' : 'src-not-cited';
    const badgeText  = isCited ? '✓引用' : '未引用';
    srcEl.innerHTML +=
      `<div><span class="${badgeClass}">${badgeText}</span>${src.title}</div>`;
  });
}

データの受け取り方

GAS の ragQueryInternal_() が extractionRate（0.0〜1.0）と extractionDetail（boolean 配列）をレスポンスに含めます。クライアント側は parseExtractionRate_() が回答本文を解析してどのソースが引用されたか判定した結果を受け取るだけで、判定ロジックは GAS 側に閉じています。

JavaScript 関数

admin UI JavaScript
let editTarget = null;

function openEditNs(preview, currentNs) {
  editTarget = preview;
  const modal = document.getElementById('editNsModal');
  const current = (currentNs || '').split(',').map(s => s.trim());

  // 既存のチェックボックスを現在のネームスペースに合わせて初期化
  modal.querySelectorAll('input[type=checkbox]').forEach(cb => {
    cb.checked = current.includes(cb.value);
  });
  modal.style.display = 'flex';
}

function closeEditNs() {
  document.getElementById('editNsModal').style.display = 'none';
  editTarget = null;
}

function saveEditNs() {
  const modal = document.getElementById('editNsModal');
  const checked = [...modal.querySelectorAll('input[type=checkbox]:checked')]
    .map(cb => cb.value);
  const newNs = checked.join(',');

  google.script.run
    .withSuccessHandler(() => { closeEditNs(); location.reload(); })
    .withFailureHandler(e => alert('保存失敗: ' + e.message))
    .adminUpdateKey(editTarget, newNs);
}

google.script.run の呼び出しパターン

google.script.run は GAS WebApp のクライアント側から GAS 関数を呼び出す API です。withSuccessHandler() と withFailureHandler() でコールバックを登録し、非同期で結果を受け取ります。サーバー側の adminUpdateKey(preview, ns) 関数が実際に PropertiesService に書き込みます。

関数引数処理
openEditNs(preview, currentNs)キー識別子、現在のNS文字列モーダルを開き、チェックボックスを初期化
closeEditNs()なしモーダルを閉じ、editTarget をリセット
saveEditNs()なし（editTarget を参照）チェック済み NS を収集し GAS に送信

タイムアウト設定

GASの処理時間（Notion検索 + Gemini API）は通常3〜8秒かかる。各UIのHTTPクライアントは30秒のタイムアウトを設定すること。それ以下にするとタイムアウトエラーが多発する。
